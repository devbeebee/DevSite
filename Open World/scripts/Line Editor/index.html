using UnityEngine;
using UnityEditor;
using System.Collections.Generic;

[CustomEditor(typeof(LineRendererRoute))]
class LineEditor : Editor
{
    public override void OnInspectorGUI()
    {
        base.OnInspectorGUI();
        if (GUILayout.Button("Populate"))
        {
            LineRendererRoute line = target as LineRendererRoute;
            line.line = line.GetComponent<LineRenderer>();
            line.transform.eulerAngles = new Vector3(90, 0, 0);
            List<Vector3> pos = new List<Vector3>();

            for (int i = 0; i < 100; i++)
            {
                pos.Add(new Vector3(0, 0, i));
            }

            line.SetPositions(pos.ToArray(), false);
        }
    }
}
